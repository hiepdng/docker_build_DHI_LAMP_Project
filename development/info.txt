HTTPD CA Certificates inside container:

mkdir -p /usr/local/share/ca-certificates/
openssl genrsa -out my-internal-ca.key 2048
openssl req -x509 -new -nodes -key my-internal-ca.key -sha256 -days 1826 -out /usr/local/share/ca-certificates/my-internal-ca.crt  -subj "/C=US/ST=state/L=City/O=Organization/OU=Department/CN=server.example.com/emailAddress=you@example.com"

# Update the certificate store to include your custom CA
apt-get update && \
apt-get install -y ca-certificates && \
update-ca-certificates

Behind the scenes on Debian, /etc/ssl/certs/ca-certificates.crt is generated by the update-ca-certificates utility.
That utility reads certificate files from /usr/share/ca-certificates/ and /usr/local/share/ca-certificates/, creates 
individually hashed symlinks in /etc/ssl/certs/, and then concatenates them all into the single ca-certificates.crt file
for applications that require a single bundle

HTTPD SSL/TLS Certificate:

# Create a self-signed SSL Certificate for testing purposes:
openssl genrsa -des3 -passout pass:YourPasswordHere -out certs/httpd/server.key.secure
openssl rsa -in certs/httpd/server.key.secure -passin pass:YourPasswordHere -out certs/httpd/server.key # decrypted server.key, used for auto start web withour password
openssl req -new -x509 -nodes -sha1 -days 365 -key certs/httpd/server.key -out certs/httpd/server.crt -extensions usr_cert  -subj "/C=US/ST=state/L=City/O=Organization/OU=Department/CN=server.example.com/emailAddress=you@example.com"

-Required enabled entries in httpd-ssl.conf
-----
<VirtualHost _default_:443>
  DocumentRoot /usr/local/apache2/htdocs
  ServerName www.example.com:443
  ServerAdmin you@example.com
  
  <Directory /usr/local/apache2/htdocs>
    Options Indexes FollowSymLinks
    AllowOverride All
    Require all granted
  </Directory>

  # Proxy PHP requests to the PHP-FPM container
  <FilesMatch \.php$>
    SetHandler "proxy:fcgi://my-php-fpm:9000"
  </FilesMatch>
</VirtualHost>
